1 & 2 & 0 \\
5 & 0 & 1 \\
0 & 2 & 4 \\
3 & -1 & 0 \\
\end{array}\right]$ and $\vc{y} =  \left[\begin{array}{r}
1 \\ 0 \\ -1 \\ 5 \end{array}\right]$.
